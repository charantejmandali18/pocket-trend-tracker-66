# AWS Credentials Template
# Copy this content to ~/.aws/credentials and fill in your actual keys

[old-account]
aws_access_key_id = YOUR_OLD_ACCOUNT_ACCESS_KEY_HERE
aws_secret_access_key = YOUR_OLD_ACCOUNT_SECRET_KEY_HERE

[xpend]
aws_access_key_id = YOUR_NEW_ACCOUNT_ACCESS_KEY_HERE
aws_secret_access_key = YOUR_NEW_ACCOUNT_SECRET_KEY_HERE

# Instructions:
# 1. Replace YOUR_OLD_ACCOUNT_ACCESS_KEY_HERE with your old account access key
# 2. Replace YOUR_OLD_ACCOUNT_SECRET_KEY_HERE with your old account secret key
# 3. Replace YOUR_NEW_ACCOUNT_ACCESS_KEY_HERE with your new account access key
# 4. Replace YOUR_NEW_ACCOUNT_SECRET_KEY_HERE with your new account secret key
